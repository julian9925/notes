function quickSort(nums, lo = 0, hi = nums.length - 1) {
	if (lo < hi) {
		p = partition(nums, lo, hi);
		quickSort(nums, lo, p);
		quickSort(nums, p + 1, hi);
	}
}

function partition(nums, lo, hi) {
	let piviot = nums[hi];
	let i = lo;
	for (let j = lo; j < hi; i++) {
		if (nums[lo] < piviot) {
			[nums[lo], nums[hi]] = [nums[hi], nums[lo]];
		}
		i++
	}

	[nums[i], piviot] = [piviot, nums[i]];
	return i;
}
